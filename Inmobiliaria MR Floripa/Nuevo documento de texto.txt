<script>
  let currentIndex = 0;
  const slides = document.querySelectorAll('.slide');
  const dotsContainer = document.getElementById('dots');
  const nextBtn = document.getElementById('next');
  const prevBtn = document.getElementById('prev');
  let autoplay = true;
  let slideInterval;

  // Crear puntos de navegación
  slides.forEach((_, i) => {
    let dot = document.createElement('span');
    dot.addEventListener('click', () => {
      showSlide(i);
      stopAutoplay();
    });
    dotsContainer.appendChild(dot);
  });
  const dots = document.querySelectorAll('#dots span');

  function showSlide(index) {
    slides[currentIndex].classList.remove('active');
    dots[currentIndex].classList.remove('active');
    currentIndex = index;
    slides[currentIndex].classList.add('active');
    dots[currentIndex].classList.add('active');
  }

  function nextSlide() {
    let index = (currentIndex + 1) % slides.length;
    showSlide(index);
  }
  function prevSlide() {
    let index = (currentIndex - 1 + slides.length) % slides.length;
    showSlide(index);
  }

  nextBtn.addEventListener('click', () => {
    nextSlide();
    stopAutoplay();
  });
  prevBtn.addEventListener('click', () => {
    prevSlide();
    stopAutoplay();
  });

  function startAutoplay() {
    slideInterval = setInterval(nextSlide, 5000);
  }
  function stopAutoplay() {
    autoplay = false;
    clearInterval(slideInterval);
  }

  // Inicializar slider
  showSlide(0);
  if (autoplay) startAutoplay();

  // Modal logic
  const modalButtons = document.querySelectorAll('.btn-detalle');
  const modals = document.querySelectorAll('.modal-bg');
  const closeButtons = document.querySelectorAll('.modal-close');

  modalButtons.forEach(btn => {
    btn.addEventListener('click', () => {
      const modalId = btn.getAttribute('data-modal');
      document.getElementById(modalId).classList.add('active');
      stopAutoplay();
    });
  });

  closeButtons.forEach(btn => {
    btn.addEventListener('click', () => {
      const modalId = btn.getAttribute('data-close');
      document.getElementById(modalId).classList.remove('active');
      // Opcional: si querés que vuelva el autoplay al cerrar modal
      // if (!autoplay) {
      //   autoplay = true;
      //   startAutoplay();
      // }
    });
  });

  // Cerrar modal al hacer click fuera del contenido
  modals.forEach(modal => {
    modal.addEventListener('click', e => {
      if(e.target === modal) {
        modal.classList.remove('active');
        // if (!autoplay) {
        //   autoplay = true;
        //   startAutoplay();
        // }
      }
    });
  });
</script>
